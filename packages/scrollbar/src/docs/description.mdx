Горизонтальная прокрутка.

```tsx live
render(() => {
const scrollbarStyle: React.CSSProperties = {
    background: 'grey',
    width: '100%',
    margin: '10px 0px',
    padding: 10,
    whiteSpace: 'nowrap',
    overflow: 'auto'
};

const childStyle: React.CSSProperties = {
  display: 'inline-block',
  width: 250,
  height: 150,
  textAlign: 'center',
  lineHeight: '150px',
  fontSize: 24
};

return (
    <Scrollbar style={scrollbarStyle} autoHide={false}>
        {new Array(10).fill(null).map((_, idx) => <div key={idx} style={childStyle}>{idx}</div>)}
    </Scrollbar>
    )
})
```

Вариант с render-prop.

```tsx live
render(() => {
    return <Scrollbar style={{height: 250}} autoHide={false}>
    {({ scrollableNodeRef, contentNodeRef, scrollableNodeClassName, contentNodeClassName }) => {
        return (
            <div ref={scrollableNodeRef} className={scrollableNodeClassName}>
                <div ref={contentNodeRef} className={contentNodeClassName}>
                    {new Array(10).fill(null).map((_, idx) => <div key={idx} style={{display: 'flex', alignItems: 'center', justifyContent: 'center', height: 100, background: 'grey', fontSize: '24px'}}>{idx}</div>)}
                </div>
            </div>
        )
    }}
</Scrollbar>
})
```
